Generar un scatterplot o diagrama de dispersión

#Importar bases de datos
data("airquality")
data("HairEyeColor")

#Generar un scatterplot o diagrama de dispersión

#Definir variables
x <- airquality$Temp
y <- airquality$Ozone

plot(x,y,
     main = "Relación entre Temperatura y Ozono",
     xlab = "Temperatura [°F]",
     ylab = "Ozono [ppb]",
     col = airquality$Month
)

Generar un gráfico de barras

#Generar un gráfico de barras 

#Promedio de Ozono por mes
df <- aggregate(Ozone ~ Month, data = airquality, FUN = mean)

#Definir colores
colores_barras <- c("#D8B5FF", "#A193E6", "#8665B8", "#894F8C", "#3F0066") 


barplot(df$Ozone,
        names.arg = df$Month,
        main = "Concentración de Ozono promedio por mes",
        xlab = "Mes",
        ylab = "Ozono [ppb]",
        ylim = c(0,70),
        col = colores_barras
)

Generar un gráfico de pastel

#Generar un gráfico de pastel

#Llevar la tabla a un dataframe para poder llamar sus columnas
df2 <- data.frame(HairEyeColor)

#Obtener la frecuencia total de cada color de ojos
df3 <- aggregate(Freq ~ Eye, data = df2, FUN = sum)

colores_barras2 <- c("#693a08", "#4bbff6", "#587631", "#8ff17d")

pie(df3$Freq,
labels = df3$Eye,
main = "Distribución de la población por color de ojos",
col = colores_barras2
)

Generar un Histograma

#Generar un Histograma

hist(
  x = airquality$Ozone, 
  breaks = 8,
  main = "Distribución de la concentración de Ozono", 
  xlab = "Ozono [ppb]", 
  ylab = "Frecuencia",
  xlim = c(0, 200),
  ylim = c(0, 40),
  col = "lightgreen"
)

Gráfico de dispersión de longitud del pico vs masa (peso) por especie

# Gráfico de dispersión de longitud del pico vs masa (peso) por especie
penguins |>
  ggplot(aes(x = bill_length_mm, y = body_mass_g)) +
  geom_point(aes(color = species,
                 shape = species),
             size = 2) +
  geom_smooth(method = "lm", se = FALSE, aes(color = species)) +
  scale_color_manual(values = c("darkorange", "darkorchid", "cyan4")) +
  ggtitle("Longitud del pico vs. masa por especie") +
  xlab("Longitud del pico (mm)") +
  ylab("Masa (g)") +
  labs(color = "Especie", shape = "Especie")
